## Ora2Pg을 이용하여 Oracle SCOTT 스키마를 PostgreSQL로 이관하는 전체 과정을 수행한다


## 19c에서 scott유저를 생성하고 권한을 부여한다
```bash
SQL> create user scott identified by tiger;

사용자가 생성되었습니다.

SQL> GRANT CREATE SESSION, RESOURCE TO SCOTT;

권한이 부여되었습니다.

SQL> ALTER USER SCOTT QUOTA UNLIMITED ON USERS;

사용자가 변경되었습니다.

```

## 생성된 테이블을 확인한다
```bash
-bash-4.2$ sqlplus -S scott/tiger@192.168.23.76:1521/orcl <<EOF
> SET PAGESIZE 50 LINESIZE 120
> SELECT table_name FROM user_tables ORDER BY 1;
>
> SELECT 'EMP'      AS tbl, COUNT(*) AS cnt FROM emp
> UNION ALL SELECT 'DEPT',     COUNT(*) FROM dept
> UNION ALL SELECT 'SALGRADE', COUNT(*) FROM salgrade
> UNION ALL SELECT 'BONUS',    COUNT(*) FROM bonus
> ORDER BY 1;
> EXIT
> EOF

TABLE_NAME
------------------------------------------------------------------------------------------------------------------------
BONUS
DEPT
EMP
SALGRADE


TBL             CNT
-------- ----------
BONUS             0
DEPT              4
EMP              14
SALGRADE          5
```

## root계정에서 기본 패키지를 설치한다
```bash
-bash-4.2$ su -
[root@orcl ~]# yum install -y perl perl-CPAN perl-DBI perl-Test-Simple make gcc wget tar yum-utils

# EPEL (perl-IPC-Run 제공)
[root@orcl ~]# yum install -y oracle-epel-release-el7

# Software Collections (libicu-devel 등)
[root@orcl ~]# yum install -y oracle-softwarecollection-release-el7
[root@orcl ~]# yum-config-manager --enable ol7_software_collections

# Optional (libedit-devel 제공)
[root@orcl ~]# yum-config-manager --enable ol7_optional_latest

# 확인: epel, software, optional 세 줄이 보여야 함
[root@orcl ~]# yum repolist | grep -iE "epel|software|optional"

#  보조 devel 패키지 설치
yum install -y perl-IPC-Run libedit-devel libicu-devel

#  postgresql15-devel 헤더만 설치 (--nodeps)
## PGDG에서 RPM만 다운로드
[root@orcl ~]# cd /tmp
[root@orcl tmp]# yumdownloader postgresql15-devel

Loaded plugins: langpacks

## LLVM 의존성 무시 (JIT용이라 DBD::Pg 빌드에 불필요)
[root@orcl tmp]# rpm -ivh --nodeps postgresql15-devel-*.rpm
```

## 헤더 생성을 확인한다
```bash
[root@orcl tmp]# ls /usr/pgsql-15/include/libpq-fe.h
[root@orcl tmp]# /usr/pgsql-15/bin/pg_config --includedir    # → /usr/pgsql-15/include
[root@orcl tmp]# /usr/pgsql-15/bin/pg_config --libdir        # → /usr/pgsql-15/lib
```

## ExtUtils::MakeMaker 업그레이드 (local::lib 방식)
```bash
# CPAN 첫 실행 시 'local::lib' 선택 → 자동으로 /root/perl5 에 최신 MakeMaker 설치
cpan ExtUtils::MakeMaker
```
- 설치 과정에서 ~/.bashrc에 아래 5줄이 자동 추가된다
```bash
PATH="/root/perl5/bin${PATH:+:${PATH}}"; export PATH;
PERL5LIB="/root/perl5/lib/perl5${PERL5LIB:+:${PERL5LIB}}"; export PERL5LIB;
PERL_LOCAL_LIB_ROOT="/root/perl5${PERL_LOCAL_LIB_ROOT:+:${PERL_LOCAL_LIB_ROOT}}"; export PERL_LOCAL_LIB_ROOT;
PERL_MB_OPT="--install_base \"/root/perl5\""; export PERL_MB_OPT;
PERL_MM_OPT="INSTALL_BASE=/root/perl5"; export PERL_MM_OPT;
```

## 현재 쉘에도 반영한다
```bash
[root@orcl tmp]# source ~/.bashrc

# 7.64 이상이 출력되어야 함 (없으면 이후 cpan -fi DBD::Pg가 실패)
[root@orcl tmp]# perl -MExtUtils::MakeMaker -e 'print $ExtUtils::MakeMaker::VERSION, "\n"'
```

## DBD::Pg 컴파일 및 설치한다
```bash
# pg_config이 PG 15를 가리키도록 PATH 설정
[root@orcl tmp]# export POSTGRES_HOME=/usr/pgsql-15
[root@orcl tmp]# export PATH=/usr/pgsql-15/bin:$PATH
[root@orcl tmp]# which pg_config              # → /usr/pgsql-15/bin/pg_config
[root@orcl tmp]# pg_config --version          # → PostgreSQL 15.x

# DBD::Pg를 PG 15 libpq와 함께 컴파일
[root@orcl tmp]# cpan -fi DBD::Pg
```

## 실제 접속 테스트를 한다
- "PostgreSQL 15.17 on x86_64-pc-linux-gnu, compiled by gcc ... | postgres | postgres" 이렇게 뜨면 성공
```bash
[root@orcl tmp]# perl -e '
> use strict;
> use DBI;
>
> my $dbh = DBI->connect(
>     "dbi:Pg:dbname=postgres;host=localhost;port=5432",
>     "postgres", "oracle",
>     { RaiseError => 1, AutoCommit => 1 }
> );
>
> my $sth = $dbh->prepare("SELECT version(), current_database(), current_user");
> $sth->execute;
> while (my @row = $sth->fetchrow_array) {
>     print join(" | ", @row), "\n";
> }
> $dbh->disconnect;
> '
```
